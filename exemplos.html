Exemplos Práticos - Estrutura `for`

1. Contar de 1 a 10

for (let i = 1; i <= 10; i++) {
    console.log(i);
}

2. Imprimir números pares de 1 a 20

for (let i = 1; i <= 20; i++) {
    if (i % 2 === 0) {
        console.log(i);
    }
}

3. Tabuada de 5

for (let i = 1; i <= 10; i++) {
    console.log(`5 x ${i} = ${5 * i}`);
}

Exemplos Práticos - Estrutura `while`

1. Soma de números até ser maior que 20

let soma = 0;
let numero = 1;
 
while (soma <= 20) {
    soma += numero;
    console.log(`Soma atual: ${soma}`);
    numero++;
}

2. Encontrar o primeiro múltiplo de 3 maior que 10

let numero = 11;
 
while (numero % 3 !== 0) {
    numero++;
}
console.log(`O primeiro múltiplo de 3 maior que 10 é: ${numero}`);

Exemplos Práticos - Estrutura `do...while`

1. Pedir um número até que seja maior que 5

let numero;
 
do {
    numero = Math.floor(Math.random() * 10);
    console.log(`Número gerado: ${numero}`);
} while (numero <= 5);

2. Pedir a idade até que seja maior que 18

let idade;
 
do {
    idade = prompt("Digite sua idade:");
} while (idade < 18);
 
console.log("Você tem mais de 18 anos!");
